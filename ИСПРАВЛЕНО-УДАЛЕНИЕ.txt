🔧 ТЕХНИЧЕСКИЕ ИЗМЕНЕНИЯ
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Добавлено поле lastPushTime в конструктор
   this.lastPushTime = 0;

2. Изменен слушатель изменений:
   БЫЛО:
     if (window.isLocalUpdate) {
         return; // Пропускаем
     }
   
   СТАЛО:
     const timeSinceLastPush = Date.now() - this.lastPushTime;
     if (timeSinceLastPush < 2000) {
         return; // Пропускаем если < 2 сек
     }

3. Изменена функция pushToCloud:
   БЫЛО:
     window.isLocalUpdate = true;
     await set(dataRef, data);
     window.isLocalUpdate = false;
   
   СТАЛО:
     this.lastPushTime = Date.now();
     await set(dataRef, data);
